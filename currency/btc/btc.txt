install docker
    1.update
    2.install apt-transport-https ca-certificates curl software-properties-common

1.download bitcoin core
    wget https://bitcoincore.org/bin/bitcoin-core-0.17.0.1/bitcoin-0.17.0.1-x86_64-linux-gnu.tar.gz
    tar -xvf

2.start
    vim /root/.bitcoin/bitcoin.conf
    ./bitcoind

3.bit docket
    3.1 run in docker
        sudo docker run -it -p 19001:19001 -p 19011:19011 freewil/bitcoin-testnet-box
        make start
    3.2 look message
        make getinfo
    3.3 产生一个区块
        make generate
    3.4 产生100个区块
        make generate BLOCKS=100
    3.5 查看节点状态
        bitcoin-cli -datadir=1  getinfo
    3.6 查看钱包地址
        bitcoin-cli -datadir=1  getwalletinfo
        bitcoin-cli -datadir=2 getnewaddress
        或者
        make address2
        bitcoin-cli -datadir=2  getnewaddress

    3.7 查看地址私钥
        bitcoin-cli -datadir=2  dumpprivkey '2NCfVMLkLQtXw7BuMD5FujscHrpfEVz1HzQ'
        bitcoin-cli -datadir=2  dumpprivkey '2MxHYWRrfhEo8CnPYc9Kr4Q3o6vJAhWogSD'
            cU2TG1JduHSataGeYHeZJ5MRgkgZttZeqYoB56sJSmDRKTyzt7Pf

    3.8 给钱包2转账
        make sendfrom1   ADDRESS=2MxHYWRrfhEo8CnPYc9Kr4Q3o6vJAhWogSD  AMOUNT=10

    3.9 查看每个地址的余额
        listaddressgroupings

        查看地址接收了多少钱
        listreceivedbyaddress
